from sklearn.linear_model import LinearRegression
X = np.atleast_2d(np.array([float(h) for h in minor.loc[:, 'Mean Years of Education']])).transpose()
y = np.array(minor.loc[:, 'Life Expectancy at Birth'])
n = X.shape[0]
test = int(0.20*n)         
perm = rn.permutation(n)   
X = X[perm]
y = y[perm]
X_test = X[:test]      
y_test = y[:test]
X_train = X[test:]    
y_train = y[test:]
reg = LinearRegression().fit(X_train, y_train)
b, m = reg.intercept_, reg.coef_
b, m
plt.plot(X_train, y_train, 'b.')
plt.plot(X_test, y_test, 'b+')
